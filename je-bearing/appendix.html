Data Schema

Entity schemas

{
  id:                   string,      // "O1" — canonical identifier
  name:                 string,      // "Joe"
  skillTier:            string,      // "Expert" | "Advanced" | "Journeyman"
  certifiedMachines:    string[],    // machines operator is qualified to run
  preferredMachines:    string[],    // machines operator runs fastest
  maxMachines:          number,      // hard ceiling on simultaneous coverage
  otEligible:           boolean,     // can be scheduled for overtime
  setupProficiency:     string,      // "Fast" | "Average" | "Slow"
  learningCurveStrength:number,      // 0–1, higher = faster setup improvement on repeat jobs
  weeklyPlannedHours:   number,      // scheduled hours this week
  weeklyActualHours:    number,      // actual hours (for actuals tracking)
  weeklySetupHours:     number,      // hours spent on setups this week
  avgMachinesCovered:   number,      // rolling average of machines simultaneously covered
  risk:                 string       // "Overload" | "Watch" | "OK" | "Under"
}
{
  id:              string,   // "VMC1" — canonical identifier
  type:            string,   // "CNC Mill" | "CNC Lathe" | "Manual Mill" etc
  ageClass:        string,   // "New" | "Mid" | "Old"
  cycleBandMin:    number,   // minimum cycle time for typical job (minutes)
  cycleBandMax:    number,   // maximum cycle time for typical job (minutes)
  setupHours:      number,   // average setup time (hours)
  precision:       string,   // "High" | "Medium" | "Low"
  babysitting:     number,   // 0–1, fraction of cycle requiring operator attention
  varRisk:         string,   // "Low" | "Medium" | "High" | "Very High"
  utilization:     number    // current utilization rate (0–1)
}
{
  id:                     string,   // "J101"
  customer:               string,   // "OEMA"
  marginClass:            string,   // "High" | "Medium" | "Low"
  qtyRemaining:           number,   // units left to produce
  dueBucket:              string,   // "This Week" | "Next Week" | "Two+ Weeks"
  setupRequired:          boolean,  // does this job need a new setup?
  preferredMachine:       string,   // ideal machine for this job
  altMachine:             string,   // acceptable fallback machine
  strategic:              boolean,  // is this customer strategically important?
  canSplit:               boolean,  // can the run be split across days/weeks?
  inventoryToleranceDays: number,   // how many days early/late is acceptable?
  idealBatch:             number,   // optimal run quantity
  lcSensitivity:          string,   // learning curve: "High" | "Medium" | "Low"
  expedite:               boolean,  // is this job currently expedited?
  urgencyScore:           number,   // 0–10 urgency score
  strategicScore:         number    // 0–10 strategic importance score
}
{
  setupCostPerHour:            number,   // $/hour for setup labor — default $85
  inventoryCarryingPct:        number,   // annual carry cost as % of part value — default 18%
  maxMachinesPerOperator:      number,   // attention ceiling — default 2.5
  batchExtensionThresholdPct:  number,   // when to extend batch to avoid 2nd setup — default 30%
  expeditePenaltyMultiplier:   number,   // cost multiplier for expedited jobs — default 1.4×
  overtimePenaltyMultiplier:   number,   // labor rate multiplier for OT — default 1.2×
  machineSubstitutionPenaltyPct: number, // cycle time penalty for using alt machine — default 12%
  attentionDilutionFactor:     number,   // cycle elongation per extra machine — default 15%
  microStopFactorMin:          number,   // min micro-stop overhead — default 5%
  microStopFactorMax:          number,   // max micro-stop overhead — default 20%
  overtimeRateMultiplier:      number,   // OT pay rate — default 1.5×
  latenessPenaltyPerDay:       number    // $ per day late — default $200
}
{
  partId:             string,      // "P201"
  partFamily:         string,      // "Bearing Ring" | "Shaft" | "Flange" etc
  material:           string,      // "4140 Steel" | "17-4 PH" | "304 SS" etc
  diameterMm:         number,      // outer diameter in mm
  lengthMm:           number,      // length in mm
  toleranceClass:     string,      // "Tight" | "Medium" | "Loose"
  machineFamily:      string,      // "CNC Lathe" | "CNC Mill" | "CNC Mill + Lathe"
  routingFamily:      string,      // "Turn-Grind-Inspect" | "Face-Mill-Drill-Inspect" etc
  setupFamily:        string,      // "Ring Soft Jaw" | "Shaft Collet" | "Box Fixture" etc
  historicalSetupH:   number,      // actual setup hours from prior runs
  historicalCycleMin: number,      // actual cycle time from prior runs
  repeatRuns:         number,      // number of times this part has been produced
  imageFamily:        string,      // "ring_like" | "shaft_like" | "flange_like" etc
  candidates:         Array<{
    partId:  string,
    score:   number,               // 0–1 similarity score
    reasons: string[]              // attribute-match explanations
  }>
}
Logic Layer

Rules registry

Scoring Formulas

Job Risk Score
score = 0
if due === "This Week"    → +4
if due === "Next Week"    → +2
if expedite === true      → +expeditePenalty
if setupRequired          → +1.5
if canSplit === false      → +1.0
if strategic === true     → -2.0
if marginClass === "High" → -1.0
clamp(0, 10)
Operator Load Score
loadRatio = actualHours / plannedHours
score = (loadRatio - 1) × 5
      + machinesCovered × 0.4
      + setupHours / 10
scaled to 0–10
Effective Cycle Time
effectiveCycle = nominalCycle
  × (1 + attentionDilution
       × (machinesCovered - 1))
  × (1 + microStopFactor)

Similarity Formula

Part Similarity Score (0–1)
score =
  0.20 × material_match
+ 0.20 × setup_family_match
+ 0.15 × routing_family_match
+ 0.15 × machine_family_match
+ 0.10 × tolerance_similarity
+ 0.10 × geometry_family_match
+ 0.10 × batch_and_history_match
Setup Savings Estimate
estimatedSavings =
  baseSetupHours
  × similarityScore
  × setupReuseFactor (0.6)

cycleConfidence =
  historicalMatchConfidence
  × machineCompatibilityFactor
